#include<stdio.h>
int main()
{
    int n,num,val1,sum=0;
    printf("Enter a number :");
    scanf("%d",&n);
    num=(n*n);
    printf("Square of given num is %d\n",num);
    while(num>0)
    {
        val1=(num%10);
        sum=sum+val1;
        num=num/10;
    }
    if(sum==n)
    {
       printf("It is a neon number");
    }
    else
    {
        printf("It is not a neon number");
    }
    return 0;
}
