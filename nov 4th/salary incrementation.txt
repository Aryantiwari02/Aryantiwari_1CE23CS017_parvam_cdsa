#include<stdio.h>
int main()
{
    int h,pay;
    int rate;
    printf("Enter hours of working:");
    scanf("%d",&h);
    printf("Enter extra amount to pay for per hour:");
    scanf("%d",&rate);
    printf("Enter base salary");
    scanf("%d",&pay);
    if(h>40)
    {
        pay=pay+(40*(rate*1.0))+((h-40)*(rate*1.5));
    }
    else{
        pay=pay+(h*(rate*1.0));
    }
    printf("Final salary:%d",pay);
}
