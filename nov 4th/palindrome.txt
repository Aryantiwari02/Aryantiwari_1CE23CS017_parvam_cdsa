#include<stdio.h>
int main()
{
    int a[5];
    int n,i;
    printf("enter  the  of length (array)value : ");
    scanf("%d",&n);
    printf("enter the array elements: \n");
    for(i=0;i<n;i++)
    {
        scanf("%d",&a[i]);
    }
    printf("Palindrome:");
    for(i=(n-1);i>=0;i--)
    {
        printf("%d\n",a[i]);
    }
    return 0;
}
